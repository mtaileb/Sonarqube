# Obtenir l'image de Sonaqube:
docker pull sonarqube:lts-community
# Lancer un conteneur Postgre pour accueillir la BDD de Sonarqube:
docker run -d --name sonarqube-db -e POSTGRES_USER=sonar -e POSTGRES_PASSWORD=sonar -e POSTGRES_DB=sonarqube postgres:alpine
# Vérifions que le conteneur s'est lancé avec:
docker ps
# Lancer Sonarqube en le liant avec la BDD Postgre:
docker run -d --name sonarqube -p 9000:9000 --link sonarqube-db:db -e SONAR_JDBC_URL=jdbc:postgresql://db:5432/sonarqube -e SONAR_JDBC_USERNAME=sonar -e SONAR_JDBC_PASSWORD=sonar sonarqube
# Vérifions que ce conteneur s'est lui aussi lancé:
docker ps
# Ensuite se connecter à SonarQube en visitant 127.0.0.1:9000 (ou localhost:9000 sur Linux)
# Ajouter le home et le path de SonarScanner ~/.bashrc:
nano ~/.bashrc
export SONAR_SCANNER_HOME=/path/to/sonar-scanner
export PATH=$SONAR_SCANNER_HOME/bin:$PATH
